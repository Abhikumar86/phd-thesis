swelev <- raster("./data/cropped-elevation.tif")

elmat <- swelev %>% raster_to_matrix()

elmat %>% sphere_shade(texture = "desert", zscale = 6, colorintensity = 5) %>%
  #height_shade() %>% 
  add_shadow(lamb_shade(elmat, zscale = 6), 0) %>%

elmat %>% height_shade() %>% 
  add_overlay(ovl1, alphalayer = 0.5) %>%
